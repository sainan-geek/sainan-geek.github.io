前言

当我们不确定有多少个参数传递的时候，可以用arguments来获取，在JavaScript中，arguments实际上是当前函数内置的一个对象。所有函数都内置了一个arguments对象，arguments对象中存储了传递的所有实参。

代码举例

```
function getMax() {
var max = arguments[0];
for (let i = 0; i < arguments.length; i++) {
if (arguments[i] > max) {
max = arguments[i];
}
}
return max;
}
var m = getMax(1, 43, 23, 65, 23, 654);
var n = getMax(2, 56, 89, 63, 354);
console.log(m);
console.log(n);

//return不能返回多个数，如果要返回多个数，返回的结果是最后一个值
function fn(sum1, sum2) {
return sum1, sum2;
}
var re = fn(1, 2);
console.log(re);

    //要返回多个结果可以返回数组
    function getre(sum1, sum2) {
        return [sum1 + sum2, sum1 - sum2, sum1 * sum2, sum1 / sum2];
    }
    var re = getre(9, 3);
    console.log(re);
```
