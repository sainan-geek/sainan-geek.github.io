对象方法相关操作

对象方法获取自身属性

Object.keys()用于获取对象属性，效果类似于for…in，返回一个由属性名称组成的数组

//定义一个手机对象
                var phone = {
            id: 1,
            pname: '小米',
            price: 1999,
            num: 2000
        };
//用于获取对象自身的属性
        var arr = Object.keys(phone);
//输出数组
        console.log(arr);
//遍历数组
        arr.forEach(function(value) {
            console.log(value);
        })

对象方法修改自身属性

Object.defineProperty(obj,prop,descriptor)定义对象中新属性或者修改原来的属性。

  • obj:必须，目标对象 (修改哪个参数)

  • prop:必须，需定义或修改属性的名字 （修改哪个属性）

  • descriptor:必须，目标属性所拥有的特性（说明）以对象的形式书写,

    descriptor中对象格式书写说明

    • value:设置默认属性的值 默认为undefined

    • writable:值是否可以重写。true｜false 默认为false

    • enumerable:目标属性是否可以枚举。true｜false 默认为false(能不能遍历出来)

    • configurable:目标属性是否可以被删除或是否可以再次修改特性。true｜false 默认为false

var phone = {
            id: 1,
            pname: '小米',
            price: 1999,
            num: 2000
        };
        //1.以前的对象属性添加和修改方式，就是直接赋值
        //phone.num = 1000;
        //phone.price = 999;
        //现在修改方式
        Object.defineProperty(phone, 'num', {
            value: 1000,
            enumerable: true,
        })

descriptor的各个参数用法举例

数据对象

var phone = {
            id: 1,
            pname: '小米',
            price: 1999,
            num: 2000
        };

设置不可修改id

        Object.defineProperty(phone, 'id', {
            writable: false,
        });
        console.log(phone);
        phone.id = 2;
        console.log(phone);

设置不可遍历属性

        Object.defineProperty(phone, 'address', {
            value: '中国吉林长春',
            writable: false,
            //如果值为false不允许遍历，默认为false
            enumerable: false,
            //如果值为false不允许修改
            configurable: false,
        });
        console.log(phone);
        console.log(Object.keys(phone));

删除属性与禁止删除

        //由于address是后添加的默认为false所以不可以删除
        delete phone.address;
        console.log(phone);
                //pname是之前就有的所以可以删除，如果要设置不可删除，进行如下操作
                /*
                Object.defineProperty(phone, 'pname', {

            configurable: true,
        });
                */
        delete phone.pname;
        console.log(phone);
